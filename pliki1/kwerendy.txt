SELECT marka, model, cena FROM pojazdy WHERE marka LIKE "BM" ORDER BY cena DESC LIMIT 15;

SELECT AVG(cena) AS "Średnia cena", MAX(cena) AS "Maksymalna cena" FROM pojazdy WHERE model LIKE "meta";

SELECT pojazdy.marka, pojazdy.model, pojazdy.cena, kolory.nazwa, kolory.doplata FROM pojazdy JOIN kolory ON pojazdy.kolor = kolory.id WHERE pojazdy.model LIKE "alfa";
